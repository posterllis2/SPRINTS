section .data
    limite_carga    dd 100          ; Valor máximo da bateria
    msg_carregando  db 'Carregando... ', 0xA
    len_carregando  equ $ - msg_carregando
    msg_concluido   db 'Carga completa. Entrando em modo Economico.', 0xA
    len_concluido   equ $ - msg_concluido

section .text
    global _start

_start:
    ; 1. Inicialização de Registradores
    mov ebx, [limite_carga]         ; Guarda o limite (100)
    mov ecx, 0                      ; Nosso sensor (0%)

loop_controle:
    ; 2. Simulação de leitura de hardware
    add ecx, 10                     ; Aumenta 10% a cada ciclo

    ; 3. Comparação
    cmp ecx, ebx                    
    jge carga_finalizada            ; Se for maior ou igual a 100, pula para o fim

    ; 4. Exibir "Carregando..." 
    mov eax, 4                      
    mov ebx, 1                     
    mov ecx, msg_carregando         
    mov edx, len_carregando         
    int 0x80                        
    
    ; Recupera o valor do sensor que estava no ECX 
    ; Para simplificar no primeiro semestre, vamos simular o sensor voltando
    mov ecx, [valor_atual]          ; (Apenas ilustrativo)
    
    ; 5. Otimização (Simulada pelo salto de volta)
    jmp loop_controle

carga_finalizada:
    ; Ação de Desligamento (Sustentabilidade)
    mov eax, 4                      
    mov ebx, 1                      
    mov ecx, msg_concluido          
    mov edx, len_concluido          
    int 0x80

encerra_processamento:
    ; Finaliza o programa
    mov eax, 1                      
    xor ebx, ebx                    
    int 0x80

section .bss
    valor_atual resd 1              ; Reserva espaço para o sensor
